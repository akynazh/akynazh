Linux实用操作之at工作调度

2022-03-31
1 minute read
Linux

at命令

create job

 1[root@VM-0-11-centos ~]# at 22:38
 2at> echo "hello world" > ~/at-test.txt
 3# ctrl D
 4at> <EOT>
 5job 3 at Thu Mar 31 22:38:00 2022
 6[root@VM-0-11-centos ~]# at now +2 minute
 7at> echo "hello kitty" >> ~/at-test.txt
 8# ctrl D
 9at> <EOT>
10job 4 at Thu Mar 31 22:39:00 2022

check job

 1[root@VM-0-11-centos ~]# ls
 2at-test.txt
 3[root@VM-0-11-centos ~]# cat at-test.txt
 4hello world
 5
 6[root@VM-0-11-centos ~]# atq # 查看将要运行的job
 74       Thu Mar 31 22:39:00 2022 a root
 8
 9[root@VM-0-11-centos ~]# cat ~/at-test.txt
10hello world
11hello kitty
12
13[root@VM-0-11-centos ~]# jobs # 查看正在进行的job情况
14[2]+  Stopped                 at now + 5 minutes

cancel job

 1[root@VM-0-11-centos ~]# at now + 1 minutes
 2at> echo "hello monkey" >> ~/at-test.txt
 3at> <EOT>
 4job 5 at Thu Mar 31 22:40:00 2022
 5[root@VM-0-11-centos ~]# atq
 65       Thu Mar 31 22:40:00 2022 a root
 7[root@VM-0-11-centos ~]# atrm 5
 8
 9after some time...
10[root@VM-0-11-centos ~]# cat ~/at-test.txt
11hello world
12hello kitty
