# 新建调度工作
[root@VM-0-11-centos ~]#crontab -e
# 进入vim界面
------
#分 时 日 月 周   
*/1  *  *  *  * test.sh
# 每一分钟执行一次
------

# 查看调度工作队列
[root@VM-0-11-centos ~]# crontab -l
*/1 * * * * test.sh
# 删除所有调度工作
[root@VM-0-11-centos ~]# crontab -r
[root@VM-0-11-centos ~]# crontab -l
no crontab for root
