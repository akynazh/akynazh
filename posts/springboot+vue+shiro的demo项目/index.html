SpringBoot Vue

前言

创建Jwt工具类

在这之前在application.yaml中配置自己的Jwt密钥和过期时间：

1auth:
2  jwt:
3    secret: dasfjlsajdlkeeawrljawr213jfsad
4    expire: 43200

注意expire以秒为单位，new Date() 时以毫秒为单位。

工具类作用如下：

  • 生成token
  • 根据用户的token获取claims用于校验token
  • 校验token是否过期
 1/**
 2 * @version 1.0
 3 * @description Jwt工具包
 4 * @Author Jiang Zhihang
 5 * @Date 2022/2/4 22:39
 6 */
 7@Data
 8@Component
 9@Slf4j
10@ConfigurationProperties(prefix = "auth.jwt")
11public class JwtUtils {
12    private String secret;
13    private int expire;
14
15    /**
16     * @description: 生成token
17     * @author Jiang Zhihang
18     * @date 2022/2/4 22:50
19     */
20    public String createToken(String email) {
21        Date nowDate = new Date();
22        Date expireDate = new Date(nowDate.getTime() + expire * 1000L); // 乘上1000ms
23        return Jwts.builder()
24                .setHeaderParam("typ", "JWT")
25                .setSubject(email)
26                .setIssuedAt(nowDate)
27                .setExpiration(expireDate)
28                .signWith(SignatureAlgorithm.HS256, secret)
29                .compact();
30    }
31
32    /**
33     * @description: 根据用户的token获取claims用于校验token
34     * @author Jiang Zhihang
35     * @date 2022/2/4 22:53
36     */
37    public Claims getClaimsByToken(String token) {
38        try {
39            return Jwts.parser()
40                    .setSigningKey(secret)
41                    .parseClaimsJws(token)
42                    .getBody();
43        } catch (Exception e) {
44            log.error("token error: {}", e.getMessage());
45            return null;
46        }
47    }
48
49    /**
50     * @description: 校验token是否过期
51     * @author Jiang Zhihang
52     * @date 2022/2/4 23:09
53     */
54    public boolean isTokenExpired(Date expireDate) {
55        return expireDate.before(new Date());
56    }
57}

创建AccountRealm类

用于向数据库查询验证账户以及权限校验信息的注册。

 1/**
 2 * @version 1.0
 3 * @description 用于向数据库查询验证账户以及权限校验信息的注册
 4 * @Author Jiang Zhihang
 5 * @Date 2022/2/4 23:21
 6 */
 7@Component
 8public class AccountRealm extends AuthorizingRealm {
 9    final JwtUtils jwtUtils;
10    final UserService userService;
11
12    @Autowired
13    public AccountRealm(JwtUtils jwtUtils, UserService userService) {
14        this.jwtUtils = jwtUtils;
15        this.userService = userService;
16    }
17
18    /**
19     * @description: 必须添加，设置支持于自己指定的JwtToken
20     * @author Jiang Zhihang
21     * @date 2022/2/5 0:31
22     */
23    @Override
24    public boolean supports(AuthenticationToken token) {
25        return token instanceof JwtToken;
26    }
27
28    /**
29     * @description: 认证校验
30     * @author Jiang Zhihang
31     * @date 2022/2/4 23:34
32     */
33    @Override
34    protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken authenticationToken) throws AuthenticationException {
35        JwtToken jwtToken = (JwtToken) authenticationToken;
36        Claims claims = jwtUtils.getClaimsByToken((String) jwtToken.getPrincipal());
37
38        String email = claims.getSubject();
39        User user = userService.getOne(new QueryWrapper<User>().eq("userEmail", email));
40        if (user == null) {
41            throw new UnknownAccountException("用户不存在");
42        }
43        // 所有对象统一为AccountProfile内置属性rid作为统一redis缓存id
44        AccountProfile profile = new AccountProfile();
45        profile.setEmail(user.getUserEmail());
46        return new SimpleAuthenticationInfo(profile, jwtToken.getCredentials(), this.getClass().getName());
47    }
48
49    /**
50     * @description: 权限校验
51     * @author Jiang Zhihang
52     * @date 2022/2/5 17:17
53     */
54    @Override
55    protected AuthorizationInfo doGetAuthorizationInfo(PrincipalCollection principalCollection) {
56        Object principal = principalCollection.getPrimaryPrincipal();
57        // 这里从redis获取的信息不能被反序列化导致无法直接强转为AccountProfile
58        AccountProfile profile = JSONUtil.parse(principal).toBean(AccountProfile.class);
59        User user = userService.getOne(new QueryWrapper<User>().eq("userEmail", profile.getEmail()));
60        SimpleAuthorizationInfo simpleAuthorizationInfo = new SimpleAuthorizationInfo();
61        simpleAuthorizationInfo.addRole(user.getUserRole());
62        java.util.List<String> permissions = Arrays.asList(user.getUserPermission().split(","));
63        simpleAuthorizationInfo.addStringPermissions(permissions);
64        return simpleAuthorizationInfo;
65    }
66}

创建Jwt过滤器

过滤流程：

  1. 首先经过prehandle，用于支持跨域请求。

  2. 接着经过onAccessDenied，拦截请求Header中是否有auth字段并进行对应操作。

  3. 如果没有token，则直接通过，后续通过@RequiresAuthentication等进行拦截。

  4. 如果有token，则校验token失效，如果失效则通过失败执行onLoginFailure。

  5. 如果token未失效，则通过Shiro进行登录认证以及可选的权限校验。

  1/**
  2 * @version 1.0
  3 * @description jwt过滤器
  4 * @Author Jiang Zhihang
  5 * @Date 2022/2/4 22:59
  6 */
  7@Component
  8public class JwtFilter extends AuthenticatingFilter {
  9    private JwtUtils jwtUtils;
 10
 11    @Autowired
 12    public void setJwtUtils(JwtUtils jwtUtils) {
 13        this.jwtUtils = jwtUtils;
 14    }
 15
 16    /**
 17     * @description: 拦截用户请求
 18     * @author Jiang Zhihang
 19     * @date 2022/2/5 0:28
 20     */
 21    @Override
 22    protected boolean onAccessDenied(ServletRequest servletRequest, ServletResponse servletResponse) throws Exception {
 23        HttpServletRequest request = (HttpServletRequest) servletRequest;
 24        String token = request.getHeader("auth");
 25        // 如果没有token，则直接通过，后续通过@RequiresAuthentication等进行拦截
 26        if (StringUtils.isEmpty(token)) {
 27            return true;
 28        } else { // 如果有token，则进行校验
 29            // 校验token
 30            Claims claims = jwtUtils.getClaimsByToken(token);
 31            if (claims == null || jwtUtils.isTokenExpired(claims.getExpiration())) {
 32                throw new ExpiredCredentialsException("token已经失效，请重新登录");
 33            }
 34            // 继续继续校验，校验成功则登录成功
 35            return executeLogin(servletRequest, servletResponse);
 36        }
 37    }
 38
 39    /**
 40     * @description: 把请求头中的token生成到JwtToken类中
 41     * @author Jiang Zhihang
 42     * @date 2022/2/5 0:28
 43     */
 44    @Override
 45    protected AuthenticationToken createToken(ServletRequest servletRequest, ServletResponse servletResponse) throws Exception {
 46        HttpServletRequest request = (HttpServletRequest) servletRequest;
 47        String token = request.getHeader("auth");
 48        if (StringUtils.isEmpty(token)) {
 49            return null;
 50        }
 51        return new JwtToken(token);
 52    }
 53
 54    /**
 55     * @description: 登录失败返回错误消息
 56     * @author Jiang Zhihang
 57     * @date 2022/2/5 17:25
 58     */
 59    @Override
 60    protected boolean onLoginFailure(AuthenticationToken token, AuthenticationException e, ServletRequest request, ServletResponse response) {
 61        HttpServletResponse httpServletResponse = (HttpServletResponse) response;
 62        Throwable throwable = e.getCause() == null ? e : e.getCause();
 63        RestfulResponse restfulResponse = RestfulResponse.fail(400, throwable.getMessage());
 64        String JsonStr = JSONUtil.toJsonStr(restfulResponse);
 65        try {
 66            httpServletResponse.getWriter().print(JsonStr);
 67        } catch (IOException ignored){
 68
 69        }
 70        return false;
 71    }
 72
 73    /**
 74     * @description: 用于支持跨域
 75     * @author Jiang Zhihang
 76     * @date 2022/2/5 0:30
 77     */
 78    @Override
 79    protected boolean preHandle(ServletRequest request, ServletResponse response) throws Exception {
 80
 81        HttpServletRequest httpServletRequest = (HttpServletRequest) request;
 82        HttpServletResponse httpServletResponse = (HttpServletResponse) response;
 83        httpServletResponse.setHeader("Access-Control-Allow-Origin", httpServletRequest.getHeader("Origin"));
 84        /*
 85        假如：浏览器地址栏的地址是：http://localhost:4200/#/pages/dashadmin
 86        异步请求后端地址：http://localhost:8080/secure
 87        请求头：
 88        ...
 89        Host:localhost:8080
 90        Origin:http://localhost:4200
 91        Referer:http://localhost:4200/
 92        ...
 93        从上面的请求头中可以看到request.getHeader("Origin")的值就是就是http://localhost:4200
 94        request.getHeader("Origin")经常用法就是用来解决浏览器跨域问题
 95        response.setHeader("Access-Control-Allow-Origin", request.getHeader("Origin"));
 96        指的是只允许http://localhost:4200跨域访问后端服务器。
 97        */
 98        httpServletResponse.setHeader("Access-Control-Allow-Methods", "GET,POST,OPTIONS,PUT,DELETE");
 99
100        /*
101        响应首部 Access-Control-Allow-Headers 用于 preflight request （预检请求）中，
102        列出了将会在正式请求的 Access-Control-Expose-Headers 字段中出现的首部信息。修改为请求首部
103         */
104        httpServletResponse.setHeader("Access-Control-Allow-Headers", httpServletRequest.getHeader("Access-Control-Request-Headers"));
105
106        // 跨域时会首先发送一个OPTIONS请求，这里我们给OPTIONS请求直接返回正常状态
107        /*
108        options 请求就是预检请求，可用于检测服务器允许的http 方法。
109        当发起跨域请求时，由于安全原因，触发一定条件时浏览器会在正式请求之前自动先发起OPTIONS 请求，即CORS 预检请求，
110        服务器若接受该跨域请求，浏览器才继续发起正式请求。
111         */
112        if (httpServletRequest.getMethod().equals(RequestMethod.OPTIONS.name())) {
113            httpServletResponse.setStatus(org.springframework.http.HttpStatus.OK.value());
114            return false;
115        }
116
117        return super.preHandle(request, response);
118    }
119}

创建Shiro配置类

这里通过@Bean，外部可以import我们自己配置的东西。

  1. 配置SessionManager

注入RedisSessionDAO，session管理员集成redis。

  1. 配置DefaultWebSecurityManager

注入AccountRealm， SessionManager，RedisCacheManager。

其中必须记得要设置主键名称，shiro-redis 插件用过这个缓存用户信息：

1redisCacheManager.setPrincipalIdFieldName("email");
  1. 配置ShiroFilterChainDefinition

在ShiroFilterChainDefinition中，我们不再通过编码形式拦截Controller访问路径，

而是所有的路由都需要经过JwtFilter这个过滤器，然后判断请求头中是否含有jwt的信息，

有就登录，没有就跳过。跳过之后，有Controller中的shiro注解进行再次拦截，比如@RequiresAuthentication，这样控制权限访问。

68    public ShiroFilterFactoryBean shiroFilterFactoryBean(org.apache.shiro.mgt.SecurityManager securityManager,
69                                                         ShiroFilterChainDefinition shiroFilterChainDefinition) {
70        ShiroFilterFactoryBean shiroBean = new ShiroFilterFactoryBean();
71        shiroBean.setSecurityManager(securityManager);
72
73        Map<String, javax.servlet.Filter> filters = new HashMap<>();
74        filters.put("jwt", jwtFilter);
75        shiroBean.setFilters(filters);
76        Map<String, String> filterChainMap = shiroFilterChainDefinition.getFilterChainMap();

 1/**

17    @ExceptionHandler(MethodArgumentNotValidException.class)

20        List<ObjectError> allErrors = e.getBindingResult().getAllErrors();
21        StringBuilder sb = new StringBuilder();
22        for (ObjectError error : allErrors) {
23            sb.append(error).append("&");
24        }

27    }

新建User控制器

注：

  1. @RequiresAuthentication代表通过过滤器后需要进一步进行认证。

  2. @RequiresRoles(“admin”)代表需要有admin对象才可进行访问。

  3. @RequiresPermissions(logical = Logical.AND, value = {“view”, “edit”})代表必须同时拥有view和edit两个权限可访问。

  4. 对于退出登录操作的解释：

退出登录操作，前端要进行删除token操作并退回登录页面，后端删除认证信息。

必须注意，如果token并没有过期，postman中前端在带着原来的token访问还是可以访问成功并生成新的认证信息。

在一次session中，用户第一次登录成功只给前端返回token，前端每次发送请求都会在请求头中携带token。

只要后端检查到请求头中的token则会进行校验，校验成功则生成一次session中的认证信息。

在这一次session中，后端保留了认证信息，这时前端即使不带token也可以访问成功，即可以成功通过@RequiresAuthentication。

但logout后删除了认证信息，不带token则访问失败。

 80     * 注意：

 82     * 2. 在一次session中，用户第一次登录成功只给前端返回token，前端每次发送请求都会在请求头中携带token，

 87     * @date 2022/2/5 17:23

 93    }

 95    /**
 96     * @description: 首次登录，无需认证，直接验证密码并生成token，并于请求头添加token
 97     * @author Jiang Zhihang
 98     * @date 2022/2/5 0:18
 99     */
100    @PostMapping("/login")
101    public RestfulResponse login(@Validated @RequestBody LoginDto loginDto, HttpServletResponse response) {
102        User myUser = userService.getOne(new QueryWrapper<User>().eq("userEmail", loginDto.getEmail()));
103        Assert.notNull(myUser, "用户不存在");
104        if (!loginDto.getPassword().equals(myUser.getUserPassword())) {
105            return RestfulResponse.fail(400, "密码错误");
106        }
107        // 密码正确则登录成功并设置token
108        String token = jwtUtils.createToken(myUser.getUserEmail());
109        response.setHeader("auth", token);
110        response.setHeader("Access-Control-Expose-Headers", "auth");
111        return RestfulResponse.success(MapUtil.builder()
112                .put("userName", myUser.getUserName())
113                .put("userEmail", myUser.getUserEmail())
114                .put("userRole", myUser.getUserRole())
115                .map()
116        );
117    }

定义axios拦截器

  1. 由于前后端分离，项目部署在不同端口上，所以设置所有axios访问后端的baseURL为后端的地址。

  2. 所有axios请求结果都先通过一个方法拦截判断成功与否，失败则reject终止并反馈信息，成功则resolve并放行，减少了许多代码量。

src/main.js

 2axios.defaults.baseURL = "http://localhost:8080"
 3axios.interceptors.response.use(
 4    response => {
 5        if (response.status === 200) {
 6            console.log(response.data)
 7            return Promise.resolve(response)
 8        } else {
 9            return Promise.reject(response)
10        }
11    },
12    error => {
13        let response = error.response
14        Message.error({
15            message: response.data.message,
16            duration: 1500,
17        })
18        return Promise.reject(response)
19    }

定义router拦截器

前后端分离，前端则必须定义自己路由的拦截器，实现资源的合理控制。

通过src/router/index.js中meta下的字段即可进行合理的操控了，比如通过auth真假与否决定一个访问静态页面资源是否需要认证。

 1// router前置拦截
 2router.beforeEach((to, from, next) => {
 3    if (to.matched.some(record => record.meta.auth)) {
 4        const token = localStorage.getItem("token")
 5        if (token) {
 6            next()
 7        } else {
 8            Element.Message({
 9                message: "请先登录哦",
10                duration: 1000
11            })
12            next({
13                path: '/login'
14            })
15        }
16    } else {
17        next()
18    }
19})
