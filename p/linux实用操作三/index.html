test命令

  • -e exist or not
  • -f file or not
  • -d directory or not
  • -r can read or not —similar: -x -w
  • -s exist and not blank
  • -nt newer than — -ot
  • -eq equal — -ne
  • -gt greater than — -lt
  • -ge greater than or equal —le
  • -z string if null -> true
  • -n string if null -> false
  • -a and
  • -o or
  • ! reverse

使用 “[]” 进行条件判断

注：中括号两端要有空白符。

[root@VM-0-11-centos tmp]# [ "a" == "b" ] && echo hello || echo bye
bye
[root@VM-0-11-centos tmp]# [ "a" == "a" ] && echo hello || echo bye
hello

read命令

  • read -a variable_name 读取数组”array”
[root@VM-0-11-centos tmp]# read -a arr
1 2 3 4 5
[root@VM-0-11-centos tmp]# echo ${arr[0]}
1
[root@VM-0-11-centos tmp]# echo ${arr[4]}
5
  • read -p “prompt words” variable_name
  • read -n limit_length variable_name
[root@VM-0-11-centos tmp]# read -n 5 kk
words[root@VM-0-11-centos tmp]# echo $kk
words
  • read -s variable_name “silent”输入不显示
  • read -t time variable_name “time”限定时间输入

条件判断语句

if else

if []; then
	do something
elif []; then
	do something
else
	do something
fi

case

case ${var} in
"condition1")
do something
;;
"condition2")
do something
;;
*) #anything
do something
;;
esac

循环语句

while

while [condition]
do
	do something
done

when meet the condition, start the loop.

until

until [condition]
do
	do something
done

when meet the condition, end the loop.

for

usage1
for $var in con1 con2 con3 ...
do
	do something
done
usage2
for $var in ${array}
...
usage3
for $var in $(seq 1 100)
...
or
for $var in {1..100}
...
usage4
for ( (i=1; i<=${var}; i++) )
...
